412  {
413  const Triangle & triangle = (*it);
414  Plane::HalfSpace location = triangle.plane().locate(vertex);
415  if (location == Plane::HalfSpace::positive)
416  {
417  removedTriangles.push_back(triangle);
418  }
419  else if (location == Plane::HalfSpace::onPlane)
420  {
421  if (triangle.isInside(vertex)) // If the vertex is inside the triangle, its contribution to the convex hull has already been taken into account.
422  {
423  return; // Nothing to do
424  }
425  }
426  }

428  for (auto it = removedTriangles.begin(), end = removedTriangles.end(); it != end; ++it)
429  {
430  removeTriangle(*it, oneReferenceEdges);
431  }
432  }

434  {
