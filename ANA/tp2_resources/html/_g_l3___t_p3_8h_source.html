191  virtual void render(double dt)
192  {
193  // We set the last dt
194  m_lastDt = (float)dt;
195  // Handles interactions with the keyboard
196  handleKeys();
197 
198  // 0 - Matrices and initialisations
199  glm::mat4 projectionMatrix = glm::perspective(glm::radians<float>(90), (float)getConfiguration().width / (float)getConfiguration().height, 0.01f, m_clippingDistance);
200  glm::mat4 viewMatrix = m_camera.getInverseTransform();
201  glm::mat4 modelMatrix = glm::mat4(1.0);
202  glm::mat3 normalMatrix = glm::transpose(glm::inverse(glm::mat3(modelMatrix)));
203  glm::vec3 lightDirection = glm::normalize(glm::vec3(0.4f, -0.3f, -1.0f));
204  glm::vec3 lightColor = glm::vec3(1, 1, 1);
205 
206  // 1 - We draw the scene
207  {
208  m_phongDirectionalShader.use();
209  // We initialize the uniforms shared by every mesh
210  m_phongDirectionalShader.trySetUniform("uni_mat_model", modelMatrix);
211  m_phongDirectionalShader.trySetUniform("uni_mat_view", viewMatrix);
212  m_phongDirectionalShader.trySetUniform("uni_mat_projection", projectionMatrix);
213  m_phongDirectionalShader.trySetUniform("uni_mat_normal", normalMatrix);
214  m_phongDirectionalShader.trySetUniform("uni_viewerPosition", m_camera.getPosition());
215  m_phongDirectionalShader.trySetUniform("uni_light.direction", lightDirection);
216  m_phongDirectionalShader.trySetUniform("uni_light.color", lightColor);
217  // We draw every mesh
218  for (int cpt = 0; cpt < m_meshes.size(); ++cpt)
219  {
220  gl3::PhongMaterial * material = m_materials[cpt];
221  material->getDiffuseTexture()->associateWithTextureUnit(0);
222  m_phongDirectionalShader.trySetUniform("uni_diffuseTexture", 0);
223  material->getSpecularTexture()->associateWithTextureUnit(1);
224  m_phongDirectionalShader.trySetUniform("uni_specularTexture", 1);
225  m_phongDirectionalShader.trySetUniform("uni_material.diffuseColor", material->getDiffuse());
226  m_phongDirectionalShader.trySetUniform("uni_material.specularColor", material->getSpecular());
227  m_phongDirectionalShader.trySetUniform("uni_material.shininess", material->getShininess());
228  gl3::VertexArrayObject * vao = &m_phongDirectionalVaos[cpt];
229  vao->bind();
230  glDrawElements(GL_TRIANGLES, vao->eboSize(), GL_UNSIGNED_INT, nullptr);
231  vao->unbind();
232  }
233  m_phongDirectionalShader.unuse();
234  }
