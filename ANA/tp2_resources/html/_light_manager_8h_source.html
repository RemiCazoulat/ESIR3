153  // We compute the camera planes oriented inside the view frustum
154  geometry::Plane frontPlane(cameraPosition, frontDirection);
155  geometry::Plane backPlane(cameraPosition, -frontDirection);
156  geometry::Plane leftPlane(cameraPosition, rightDirection);
157  geometry::Plane rightPlane(cameraPosition, -rightDirection);
158  geometry::Plane upPlane(cameraPosition, -upDirection);
159  geometry::Plane downPlane(cameraPosition, upDirection);
160  // Function that tests if a sphere intersects the positive half space
161  auto intersectPositiveHalfSpace = [](const geometry::Plane & plane, const glm::vec3 & position, float radius)
162  {
163  return plane.signedDistance(position) >= -radius;
164  };
165  // Function that tests if a sphere intersects a plane
166  auto intersectPlane = [](const geometry::Plane & plane, const glm::vec3 & position, float radius)
167  {
168  return fabs(plane.signedDistance(position)) <= radius;
169  };
170  // Function that tests if a sphere intersects the frustum
171  auto intersectFrustum = [&](const glm::vec3 & position, float radius)
172  {
173  return intersectPositiveHalfSpace(frontPlane, position, radius) &&
174  intersectPositiveHalfSpace(backPlane, position, radius) &&
175  intersectPositiveHalfSpace(leftPlane, position, radius) &&
176  intersectPositiveHalfSpace(rightPlane, position, radius) &&
177  intersectPositiveHalfSpace(upPlane, position, radius) &&
178  intersectPositiveHalfSpace(downPlane, position, radius);
179  };
180  // Classification of the lights
181  for (size_t cpt = 0, size = m_pointLights.size(); cpt < size; ++cpt)
182  {
183  const PointLight & light = m_pointLights[cpt];
184  glm::vec3 position = light.getWorldPosition();
185  float radius = light.getRadius();
186  if (!intersectFrustum(position, radius))
187  {
188  bool front = intersectPlane(frontPlane, position, radius);
189  bool back = intersectPlane(backPlane, position, radius);
190  if (front && back) // The light intersect bith planes
191  {
192  frontAndBackIntersection.push_back(cpt);
193  }
194  else if (front)
195  {
196  frontIntersection.push_back(cpt);
197  }

201  }
