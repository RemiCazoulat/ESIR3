442  std::shared_ptr<Agent> nearest; // Who is the collider
443 
444  // We search for the collision which is the closest in time.
445  for (auto it = neighbours.begin(), end = neighbours.end(); it != end; ++it)
446  {
447  const std::shared_ptr<Agent> & other = (*it);
448  float collisionTime = std::numeric_limits<float>::max();
449  bool col = canCollide(*other, timeWindow, collisionTime);

455  }
456  }

464  return separation(neighbours, adaptationTime);

491  }

493  break;
