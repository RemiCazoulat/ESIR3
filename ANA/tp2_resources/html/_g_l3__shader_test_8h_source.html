105  int solvePolynomial(glm::vec3 factors, float result, glm::vec2 & roots)
106  {
107  if (factors.x == 0.0f && factors.y == 0.0f) { return 0; }
108  if (factors.x == 0.0f)
109  {
110  float root = (result - factors.z) / factors.y;
111  roots = glm::vec2(root, root);
112  return 1;
113  }
114 
115  glm::vec3 poly = factors - glm::vec3(0.0, 0.0, result);
116  float delta = poly.y*poly.y - 4.0f*poly.x*poly.z;
117  if (delta < 0.0) { return 0; }
118  if (delta == 0)
119  {
120  float r = -poly.y / (2.0f*poly.x);
121  roots = glm::vec2(r, r);
122  return 1;
123  }
124  roots = glm::vec2((-poly.y + sqrt(delta))/ (2.0f*poly.x), (-poly.y - sqrt(delta)) / (2.0f*poly.x));
125  return 2;
126  }

128  bool distance(const glm::vec3 & diffuse, const glm::vec3 & specular, const glm::vec3 & polynomial, float threshold, float & root)
129  {
130  glm::vec3 max = glm::max(diffuse, specular);
