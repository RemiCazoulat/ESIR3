254  {
255  glm::mat4 projectionMat = glm::perspective(45.0f, (float)getConfiguration().width / (float)getConfiguration().height, 0.1f, 10000.0f);
256  glm::mat4 modelMat(1.0);
257  glm::mat4 viewMat = m_camera.getInverseTransform();
258  glm::mat3 normalMat = glm::transpose(glm::inverse(glm::mat3(modelMat)));
259 
260  m_phongShaderInstanced.use();
261 
262  glm::vec3 lightPosition(0.0, 0.0, m_geometryExtent.max().z*2.0);
263  glm::vec3 lightColor(1.0, 1.0, 1.0);
264  m_phongShaderInstanced.setUniform("uni_light.position", lightPosition);
265  m_phongShaderInstanced.setUniform("uni_light.color", lightColor);
266  m_phongShaderInstanced.setUniform("uni_mat_view", viewMat);
267  m_phongShaderInstanced.setUniform("uni_mat_projection", projectionMat);
268  m_phongShaderInstanced.setUniform("uni_viewerPosition", m_camera.getPosition());
269 
270  for (size_t cpt = 0; cpt < m_meshes.size(); ++cpt)
271  {
272  // 1 - we setup the material
273  const gl3::PhongMaterial * material = m_meshes[cpt].second;
274  //m_phongShader.setUniform("uni_material.emissiveColor", material->getEmissive());
275  //m_phongShader.setUniform("uni_material.ambientColor", material->getAmbient());
276  m_phongShaderInstanced.setUniform("uni_material.diffuseColor", material->getDiffuse());
277  m_phongShaderInstanced.setUniform("uni_material.specularColor", material->getSpecular());
278  m_phongShaderInstanced.setUniform("uni_material.shininess", material->getShininess());
279  material->getDiffuseTexture()->associateWithTextureUnit(0);
280  m_phongShaderInstanced.setUniform("uni_diffuseTexture", 0);
281  material->getSpecularTexture()->associateWithTextureUnit(1);
282  m_phongShaderInstanced.setUniform("uni_specularTexture", 1);
283  // 2 - we draw the geometry

286  ++drawCalls;
287  nbTriangles += (m_vaos[cpt].eboSize() / 3)*m_instancesTransforms.size();
288  m_vaosInstanced[cpt].unbind();
289  }
