386  static Quaternion slerp(Quaternion v0, Quaternion v1, double t) {
387  // Only unit quaternions are valid rotations. Normalize to avoid undefined behavior.
388  v0.normalize();
389  v1.normalize();
390  // Compute the cosine of the angle between the two vectors.
391  double dot = v0.s()*v1.s()+v0.v()*v1.v();// dot_product(v0, v1);
392  // If the dot product is negative, slerp won't take
393  // the shorter path. Note that v1 and -v1 are equivalent when
394  // the negation is applied to all four components. Fix by
395  // reversing one quaternion.
396  if (dot < 0.0f) {
397  v1 = -v1;
398  dot = -dot;
399  }
400 
401  const Float DOT_THRESHOLD = 0.9995;
402  if (dot > DOT_THRESHOLD) {
403  // If the inputs are too close for comfort, linearly interpolate and normalize the result.
404  Quaternion result = v0 + (v1 - v0) * t;
405  result.normalize();
406  return result;
407  }
408  // Since dot is in range [0, DOT_THRESHOLD], acos is safe
409  Float theta_0 = acos(dot); // theta_0 = angle between input vectors
410  Float theta = theta_0 * t; // theta = angle between v0 and result
411  Float sin_theta = sin(theta); // compute this value only once
412  Float sin_theta_0 = sin(theta_0); // compute this value only once
413 
414  Float s0 = cos(theta) - dot * sin_theta / sin_theta_0; // == sin(theta_0 - theta) / sin(theta_0)
415  Float s1 = sin_theta / sin_theta_0;
416 
417  return (v0*s0) + (v1*s1);
