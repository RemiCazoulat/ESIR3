718  bool update(float dt)
719  {
720  // We update the internal clock
721  m_internalClock += dt ;
722  bool updated = m_internalClock>m_internalPeriod ;
723  // If links have been modified, we sort and ensure the uniqueness of the link between two masses
724  updateLinks();
725  // We update the system using the internal update frequency
726  while(m_internalClock>m_internalPeriod)
727  {
728  // 1 - We reset forces
729  resetForces();
730  // 2 - We call modifiers
731  applyModifiers();
732  // 3 - We copy the current state to the next state (this ensure a globally consistent state)
733  m_masses.next() = m_masses.current() ;
734  // 3 - We integrate on the current state and write the results in the next state
735  m_integrator(m_internalPeriod) ;
736  // 4 - We apply position constraints by using current and next state
737  applyPositionConstraints();
738  // 5 - We commit the result: next state becomes current state
739  m_masses.commit() ;
740  // 6 - We update the internal clock
